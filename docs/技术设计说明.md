```csharp
// Assets/Scripts/Game/Data/CardData.cs
using System.Collections.Generic;
using UnityEngine;

[System.Serializable]
public class Segment
{
    public float duration;           // 片段时长
    public bool isInterruptible;     // 是否可打断

    public Segment(float duration, bool isInterruptible)
    {
        this.duration = duration;
        this.isInterruptible = isInterruptible;
    }
}

[System.Serializable]
public class CardData
{
    public int id;                   // 卡牌ID
    public string name;              // 卡牌名称
    public string description;       // 卡牌描述
    public string iconPath;          // 图标资源路径
    public List<Segment> segments;   // 读条片段
    public int panicDelta;           // 完成时慌乱值变化
    public int exciteDelta;          // 完成时兴奋值变化
    public int interruptPanicAdd;    // 打断时增加慌乱值

    public float CalculateTotalDuration()
    {
        float total = 0f;
        foreach (var seg in segments)
            total += seg.duration;
        return total;
    }
}

public class CardInstance
{
    public CardData data;
    public int instanceId;
    public bool isUsed;
    public bool isUsedSuccess;
    public float currentReadTime;
    public int currentSegmentIndex;

    public CardInstance(CardData data, int instanceId)
    {
        this.data = data;
        this.instanceId = instanceId;
        this.isUsed = false;
        this.currentReadTime = 0f;
        this.currentSegmentIndex = 0;
    }

    public Segment GetCurrentSegment()
    {
        if (currentSegmentIndex < data.segments.Count)
            return data.segments[currentSegmentIndex];
        return null;
    }

    public bool CanInterrupt()
    {
        Segment seg = GetCurrentSegment();
        return seg != null && seg.isInterruptible;
    }
}
```

```csharp
using UnityEngine;

[CreateAssetMenu(fileName = "CardConfig", menuName = "Game/Card Config")]
```

### 2.3 情绪值系统

```csharp
// Assets/Scripts/Game/Emotion/EmotionSystem.cs
using UnityEngine;
using Framework;
using Framework.EventCenter;

public class EmotionSystem : SingletonAutoMono<EmotionSystem>
{
    [SerializeField] private int panicValue;
    [SerializeField] private int exciteValue;

    private int minPanic = 30, maxPanic = 100;
    private int minExcite = 30, maxExcite = 100;
    private int criticalValue;

    public void Initialize(LevelConfigSO levelConfig)
    {
        panicValue = levelConfig.initialPanic;
        exciteValue = levelConfig.initialExcite;
        criticalValue = levelConfig.criticalValue;
        NotifyEmotionChanged();
    }

    public void ChangePanic(int delta)
    {
        panicValue = Mathf.Clamp(panicValue + delta, minPanic, maxPanic);
        NotifyEmotionChanged();
    }

    public void ChangeExcite(int delta)
    {
        exciteValue = Mathf.Clamp(exciteValue + delta, minExcite, maxExcite);
        NotifyEmotionChanged();
    }

    public bool IsCaughtByCriticalValue()
    {
        return (panicValue + exciteValue) >= criticalValue;
    }

    private void NotifyEmotionChanged()
    {
        var info = new EmotionInfo
        {
            panicValue = this.panicValue,
            exciteValue = this.exciteValue,
            totalValue = this.panicValue + this.exciteValue,
            criticalValue = this.criticalValue
        };
        EventCenter.Instance.EventTrigger(E_EventType.EmotionChanged, info);
    }
}

public struct EmotionInfo
{
    public int panicValue;
    public int exciteValue;
    public int totalValue;
    public int criticalValue;
}
```

### 2.4 老师AI

```csharp
// Assets/Scripts/Game/AI/TeacherAI.cs
using UnityEngine;
using Framework;
using Framework.EventCenter;

public enum TeacherState
{
    Idle,
    Approaching,
    Inspecting,
    Leaving
}

public enum InspectType
{
    Eye,
    Flash
}

public class TeacherAI : MonoBehaviour
{
    [SerializeField] private TeacherState currentState;
    [SerializeField] private InspectType currentInspectType;
    [SerializeField] private float stateTimer;
    [SerializeField] private int patrolCount;

    private LevelConfigSO levelConfig;

    public void Initialize(LevelConfigSO config)
    {
        levelConfig = config;
        EnterState(TeacherState.Idle);
    }

    private void Update()
    {
        if (stateTimer > 0)
        {
            stateTimer -= Time.deltaTime;
            if (stateTimer <= 0)
                OnStateTimerComplete();
        }
    }

    private void EnterState(TeacherState newState)
    {
        ExitState(currentState);
        currentState = newState;

        switch (newState)
        {
            case TeacherState.Idle:
                float interval = Random.Range(levelConfig.patrolIntervals.x, levelConfig.patrolIntervals.y);
                stateTimer = interval;
                break;

            case TeacherState.Approaching:
                float duration = Random.Range(levelConfig.patrolTime.x, levelConfig.patrolTime.y);
                stateTimer = duration;
                float flashProb = Mathf.Min(0.6f, 0.3f + patrolCount * 0.05f);
                currentInspectType = Random.value < flashProb ? InspectType.Flash : InspectType.Eye;
                EventCenter.Instance.EventTrigger(E_EventType.TeacherFootstepStart, currentInspectType);
                break;

            case TeacherState.Inspecting:
                float checkDuration = currentInspectType == InspectType.Eye
                    ? Random.Range(levelConfig.eyeCheckDuration.x, levelConfig.eyeCheckDuration.y)
                    : Random.Range(levelConfig.flashCheckDuration.x, levelConfig.flashCheckDuration.y);
                stateTimer = checkDuration;
                EventCenter.Instance.EventTrigger(E_EventType.TeacherInspectStart, currentInspectType);
                break;

            case TeacherState.Leaving:
                stateTimer = Random.Range(1.5f, 2.5f);
                EventCenter.Instance.EventTrigger(E_EventType.TeacherInspectEnd);
                break;
        }

        EventCenter.Instance.EventTrigger(E_EventType.TeacherStateChanged, newState);
    }

    private void OnStateTimerComplete()
    {
        switch (currentState)
        {
            case TeacherState.Idle:
                patrolCount++;
                EnterState(TeacherState.Approaching);
                break;

            case TeacherState.Approaching:
                EnterState(TeacherState.Inspecting);
                break;

            case TeacherState.Inspecting:
                EnterState(TeacherState.Leaving);
                break;

            case TeacherState.Leaving:
                EnterState(TeacherState.Idle);
                break;
        }
    }

    private void ExitState(TeacherState state)
    {
        // 状态退出时的清理工作
    }

    public TeacherState GetCurrentState() => currentState;
    public InspectType GetCurrentInspectType() => currentInspectType;
}
```

### 3.1 事件中心

```csharp
// 发送事件
EventCenter.Instance.EventTrigger(E_EventType.CardReadComplete);
EventCenter.Instance.EventTrigger(E_EventType.EmotionChanged, emotionInfo);

// 监听事件
EventCenter.Instance.AddEventListener(E_EventType.CardReadComplete, OnCardReadComplete);

EventCenter.Instance.RemoveEventListener(E_EventType.CardReadComplete, OnCardReadComplete);
```
